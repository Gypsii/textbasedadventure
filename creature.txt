ATTACK:
	DAMAGE 			int, the base damage of the attack
	DAMAGE_TYPE 	string, the damage type (blunt, slash, pierce, burn, cold, magic)
	VERB 			string, the verb displayed when the attack is used - defaults to verb generated by damage type.
	DURATION 		double, the amount of time after this attack before the next action - defaults to 1.

CREATURE:
	NAME  			string, name of creature
	DEF_ARTICLE 	string, definite article to address creature - defaults to "a"
	INDEF_ARTICLE  	string, indefinite article to address creature - defaults to "the"
	HP  			int, maximum hp
	RESIST_BLUNT  	int, resistance to blunt damage - defaults to 0
	RESIST_SLASH  	int, resistance to slash damage - defaults to 0
	RESIST_PIERCE  	int, resistance to pierce damage - defaults to 0
	RESIST_BURN  	int, resistance to burn damage - defaults to 0
	RESIST_COLD  	int, resistance to cold damage - defaults to 0
	RESIST_MAGIC  	int, resistance to magic damage - defaults to 0
	CRIT  			double, critical hit chance - defaults to 0
	CRIT_DMG  		double, critical hit damage multiplier - defaults to 1.25
	XP  			int, experience given when killed - defaults to 0
	DESCRIPTION  	string, description
	ATTACK  		attackId, attack used
	ITEM  			itemId (+ double), item in inventory (second argument gives chance of addition)
	BUTCHER_DROP  	itemId + double, drop when butchered; chance of success
	CAN_BE_BOSS  	boolean, whether this creature can be modified as a boss - defaults to true
	ENRAGEABLE  	boolean, whether this creature can be enraged - defaults to false
	HOSTILE 	 	boolean, whether this creature is hostile to the player - defaults to true
	ALIVE  			boolean, whether this creature is alive - defaults to true
